struct ArcShape: StyledShape {
    var startAngle: Double
    var endAngle: Double
    var clockwise: Bool

    var strokeColor: SwiftCrossUI.Color? = SwiftCrossUI.Color.black
    let fillColor: SwiftCrossUI.Color? = nil
    let strokeStyle: StrokeStyle? = StrokeStyle(width: 5.0)

    func path(in bounds: Path.Rect) -> Path {
        Path()
            .addArc(
                center: bounds.center,
                radius: min(bounds.width, bounds.height) / 2.0 - 2.5,
                startAngle: startAngle,
                endAngle: endAngle,
                clockwise: clockwise
            )
    }

    func size(fitting proposal: SIMD2<Int>) -> ViewSize {
        let diameter = Double(max(11, min(proposal.x, proposal.y)))
        return ViewSize(diameter, diameter)
    }
}

class EventWithParam<T> {
    let className: String
    private let name = NSNotification.Name(UUID().uuidString)

    init(_ className: String) {
        self.className = className
    }

   @discardableResult
    func listen(_ cb: @escaping (T) -> Void) -> () -> Void {
        let observer = NotificationCenter.default.addObserver(forName: name, object: nil, queue: nil) { obj in
            print("listening in \(self.className)")
            cb(obj.object as! T)
        }
        return { NotificationCenter.default.removeObserver(observer) }
    }
    
    func emit(_ obj: T) {
        NotificationCenter.default.post(name: name, object: obj)
    }
}

class Event {
    let className: String
    private let name = NSNotification.Name(UUID().uuidString)
    
    init(_ className: String) {
        self.className = className
    }

    @discardableResult
    func listen(_ cb: @escaping () -> Void) -> () -> Void {
        let observer = NotificationCenter.default.addObserver(forName: name, object: nil, queue: nil) { _ in
            print("listen in \(self.className)")
            cb()
        }
        return { NotificationCenter.default.removeObserver(observer) }
    }

    func emit() {
        NotificationCenter.default.post(name: name, object: nil)
    }
}
